```
from pwn import *

server_ip = '127.0.0.1'
server_port = 26582

connection = remote(server_ip, server_port)

received_data = connection.recv().decode('utf-8')
send_data = received_data[-32:-26] + received_data[-23:-17]

print(received_data)

connection.sendline(send_data.encode('utf-8'))
connection.interactive()
connection.close()
```
